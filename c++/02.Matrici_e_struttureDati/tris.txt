void stampa_mappa(vector<vector<char>> &matrice){
   
    
    for(int i = 0; i < matrice.size(); i++){
        for(int j = 0; j < matrice[i].size(); j++){
            cout << matrice[i][j] << " ";
        }
        cout << endl;
    }
}

bool scansiona_matrice(vector<vector<char>> &matrice){
// scansiona la matruce per vedere quale simbolo stampare nella casella scelta
    bool x = false;
    int conta_x = 0;
    int conta_o = 0;

    for(int i = 0; i < matrice.size(); i++){
        for(int j = 0; j < matrice[i].size(); j++){
            if(matrice[i][j] == 'x'){
                conta_x += 1;
            }else if(matrice[i][j] == 'o'){
                conta_o += 1;
            }
        }
    }

    if(conta_x == conta_o){ // se sono uguali vuol dire che tocca alla x essere stampata e quindi x diventerà true
        x = true;
    }

    return x;
}

bool verifica_vittoria(vector<vector<char>> &matrice){

    bool vittoria = false;

    if(matrice[0][0] == matrice [0][1] && matrice[0][1] == matrice[0][2]){
        vittoria = true;
    }else if(matrice[1][0] == matrice [1][1] && matrice[1][1] == matrice[1][2]){
        vittoria = true;
    }else if(matrice[2][0] == matrice [2][1] && matrice[2][1] == matrice[2][2]){
        vittoria = true;
    }else if(matrice[0][0] == matrice [1][0] && matrice[1][0] == matrice[2][0]){
        vittoria = true;
    }else if(matrice[0][1] == matrice [1][1] && matrice[1][1] == matrice[2][1]){
        vittoria = true;
    }else if(matrice[0][2] == matrice [1][2] && matrice[1][2] == matrice[2][2]){
        vittoria = true;
    }else if(matrice[0][0] == matrice [1][1] && matrice[1][1] == matrice[2][2]){
        vittoria = true;
    }else if(matrice[0][2] == matrice [1][1] && matrice[1][1] == matrice[2][0]){
        vittoria = true;
    }
    

    return vittoria;
}
