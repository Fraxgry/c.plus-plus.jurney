void pulisci_schermo() {

        system("cls");

void stampa_mappa(vector<vector<int>> &matrice){
   
    
    for(int i = 0; i < matrice.size(); i++){
        for(int j = 0; j < matrice[i].size(); j++){
            cout << matrice[i][j] << " ";
        }
        cout << endl;
    }
}

int scansiona_nemici(vector<vector<int>> &matrice){
   
    int nemici = 0;
   
    for(int i = 0; i < matrice.size(); i++){
        for(int j = 0; j < matrice[i].size(); j++){
            if(matrice[i][j] == 4){
                nemici += 1;
            }
        }
    }
    return nemici;
}

int scansiona_tesori(vector<vector<int>> &matrice){
   
    int tesori = 0;
   
    for(int i = 0; i < matrice.size(); i++){
        for(int j = 0; j < matrice[i].size(); j++){
            if(matrice[i][j] == 5){
                tesori += 1;
            }
        }
    }
    return tesori;
}

                    eroe.y -= 1;
                    mappa[eroe.x][eroe.y] = 9;

               if( eroe.x + 1 >= mappa.size() || mappa[eroe.x+1][eroe.y] == 1){
                    cout << "Strada bloccata!!" << endl;

                    if(mappa[eroe.x + 1][eroe.y] == 5){
                        eroe.monete += 1;
                        cout << "Tesoro trovato!!" << endl;

                        cout << "Hai trovato l'uscita!!" << endl;
                        cout << "Hai trovato: " << eroe.monete<< " monete" << endl;
                        return true;

                    mappa[eroe.x][eroe.y] = 0;
                    eroe.x += 1;
                    mappa[eroe.x][eroe.y] = 9;
                }
                
                break;

            case 'd':

               if( eroe.y + 1 >= mappa[eroe.x].size() || mappa[eroe.x][eroe.y +1] == 1){
                    cout << "Strada bloccata!!" << endl;

                    mappa[eroe.x][eroe.y] = 9;

                return true;
                break;
