/**
 * @file cacciatore_tesori.cpp
 * @brief Scorrimento di una matrice per il calcolo di un totale.
 * * Il programma inizializza una griglia contenente valori numerici che
 * rappresentano monete d'oro. Tramite una funzione dedicata, scansiona
 * l'intera matrice per sommare i valori e restituire il bottino totale.
 * * @return int Restituisce 0 in caso di esecuzione corretta.
 */

#include <iostream>
#include <vector>

using namespace std;

int somma(const vector<vector<int>> &matrice){
    
    int risultato = 0;
    
    for(int i = 0; i < matrice.size(); i++){
        for(int j = 0; j < matrice[i].size(); j++){
            risultato += matrice[i][j];
        }
    }
    return risultato;
}

int main(){
    vector<vector<int>> tesori = {{2, 4, 3}, {5, 7, 3}, {1, 5, 10}};
    
    int risultato = somma (tesori);
    cout << risultato;
    
    return 0;
}
