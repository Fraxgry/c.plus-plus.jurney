/**
 * @file mappa_grafica.cpp
 * @brief Trasformazione di dati numerici in simboli grafici.
 * * Il programma scorre la matrice 3x3 e, invece di stampare i numeri 0 e 1,
 * utilizza un controllo condizionale per stampare '#' al posto degli 1 (muri)
 * e '.' al posto degli 0 (strada), creando un dungeon visivo.
 * * @return int Restituisce 0 in caso di esecuzione corretta.
 */

#include <iostream>
#include <vector>

using namespace std;

int main(){
     vector<vector<int>> mappa = {{1, 0, 1}, {0, 0, 1}, {1, 1, 0}};
     
     for(int i = 0; i < mappa.size(); i++){
         for(int j = 0; j < mappa[i].size(); j++){
             if(mappa[i][j] == 0){
                 cout << ". ";
             }else{
                 cout << "# ";
             }
         }
         cout << endl;
     }
    
    return 0;
}
