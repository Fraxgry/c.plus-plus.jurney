 * funzioni di stampa e scansione per riferimento, logica di movimento interattiva,

#include <iostream>
#include <vector>

using namespace std;

struct Personaggio{
    
    string nome;
    int monete = 0;
    int x;
    int y;
};

void stampa_mappa(vector<vector<int>> &matrice){
    
    for(int i = 0; i < matrice.size(); i++){
        for(int j = 0; j < matrice[i].size(); j++){
            cout << matrice[i][j] << " ";
        }
        cout << endl;
        
    }
}


void scansiona_tesori(vector<vector<int>> &matrice){
    
    int contatore = 0;
    
    for(int i = 0; i < matrice.size(); i++){
        for(int j = 0; j < matrice[i].size(); j++){
            if(matrice[i][j] == 5){
                cout << "Tesoro trovato a: " << i << ", " << j << endl;
                contatore += 1;
            }
        }
    }
    cout << "Totale tesori: " << contatore << endl;
}

int main(){
    
    Personaggio eroe;
    
    vector<vector<int>> mappa = {
    {1, 0, 5, 0, 3}, // Il 3 è l'uscita
    {1, 1, 0, 1, 1},
    {0, 5, 0, 0, 0},
    {5, 1, 1, 1, 0},
    {9, 0, 5, 0, 0}  // Il 9 parte qui
};
    char scelta;
    eroe.monete= 0;
    eroe.x = 4;
    eroe.y = 0;
    
    cout << "Benvenuto nel mio daungeon. Trova l'uscita e raccogli più tesori possibile!!" << endl;
    cout << "===========COMANDI===========" << endl;
    cout << "| -W per andare in su       |" << endl;
    cout << "| -A per andare a sinistra  |" << endl;
    cout << "| -S per andare in giù      |" << endl;
    cout << "| -D per andare a destra    |" << endl;
    cout << "| -Q per uscire             |" << endl;
    cout << "=============================" << endl;
    cout << "Ciao eroe, come ti chiami?: " << endl;
    cin >> eroe.nome;
    cout << eroe.nome << ", che bel nome. Scappa dal daungeon!!" << endl;
    
    cout << endl;
    cout << endl;
    
    scansiona_tesori(mappa);
    cout << endl;
    stampa_mappa(mappa);
    
    do{
        
        cin >> scelta;
        
        switch(scelta){
            
            case 'w':
                if( eroe.x - 1 < 0 || mappa[eroe.x-1][eroe.y] == 1){
                    cout << "Strada bloccata!!\n" << endl;
                }else{
                    if(mappa[eroe.x-1][eroe.y] == 5){
                        eroe.monete += 1;
                        cout << "Tesoro trovato!!\n" << endl;
                    }else if(mappa[eroe.x-1][eroe.y] == 3){
                        cout << "Hai trovato l'uscita!!" << endl;
                        cout << "Hai trovato: " << eroe.monete<< " monete";
                        return 0;
                    }
                    mappa[eroe.x][eroe.y] = 0;
                    eroe.x -= 1;
                    mappa[eroe.x][eroe.y] = 9;
                }
                stampa_mappa(mappa);
                
                break;
                
            case 's':
                if( eroe.x + 1 >4 || mappa[eroe.x+1][eroe.y] == 1){
                    cout << "Strada bloccata!!\n" << endl;
                }else{
                    if(mappa[eroe.x+1][eroe.y] == 5){
                        eroe.monete += 1;
                        cout << "Tesoro trovato!!\n" << endl;
                    }else if(mappa[eroe.x+1][eroe.y] == 3){
                        cout << "Hai trovato l'uscita!!" << endl;
                        cout << "Hai trovato: " << eroe.monete << " monete";
                        return 0;
                    }
                    mappa[eroe.x][eroe.y] = 0;
                    eroe.x += 1;
                    mappa[eroe.x][eroe.y] = 9;
                }
                stampa_mappa(mappa);
                
                break;
                
            case 'a':
                if( eroe.y - 1 < 0 || mappa[eroe.x][eroe.y-1] == 1){
                    cout << "Strada bloccata!!\n" << endl;
                }else{
                    if(mappa[eroe.x][eroe.y - 1] == 5){
                        eroe.monete += 1;
                        cout << "Tesoro trovato!!\n" << endl;
                    }else if(mappa[eroe.x][eroe.y-1] == 3){
                        cout << "Hai trovato l'uscita!!" << endl;
                        cout << "Hai trovato: " << eroe.monete << " monete";
                        return 0;
                    }
                    mappa[eroe.x][eroe.y] = 0;
                    eroe.y -= 1;
                    mappa[eroe.x][eroe.y] = 9;
                }
                stampa_mappa(mappa);
                
                break;
                
            case 'd':
                if( eroe.y + 1 > 4 || mappa[eroe.x][eroe.y+1] == 1){
                    cout << "Strada bloccata!!\n" << endl;
                }else{
                    if(mappa[eroe.x][eroe.y + 1] == 5){
                        eroe.monete += 1;
                        cout << "Tesoro trovato!!\n" << endl;
                    }else if(mappa[eroe.x][eroe.y + 1] == 3){
                        cout << "Hai trovato l'uscita!!" << endl;
                        cout << "Hai trovato: " << eroe.monete << " monete";
                        return 0;
                    }
                    mappa[eroe.x][eroe.y] = 0;
                    eroe.y += 1;
                    mappa[eroe.x][eroe.y] = 9;
                }
                stampa_mappa(mappa);
                
                break;
                
            case 'q':
                cout << "Sto uscendo dal gioco" << endl;
                return 0;
                
            default:
                cout << "Comando non utilizzabile" << endl;
        }
        
        
    }while(scelta != 'q');
    
    return 0;
}
