/**
 * @file identikit_testo.cpp
 * @brief Unisce due parole inserite dall'utente in un'unica frase e ne calcola la lunghezza totale.
 * @param string p1 Prima parola.
 * @param string p2 Seconda parola.
 * @return int Restituisce 0 alla fine dell'esecuzione.
 */
 
#include <iostream>
#include <string>

using namespace std;

int main(){
    
    string p1;
    string p2;
    
    cout << "Inserisci la prima parola: ";
    cin >> p1;
    cout << "Inserisci la seconda parola: ";
    cin >> p2;
    
    string frase = p1 + " " + p2;
    
    int lunghezza = frase.length();
    
    cout << "La lunghezza totale delle due parole é: " << lunghezza << endl;
    
    
    return 0;
}
