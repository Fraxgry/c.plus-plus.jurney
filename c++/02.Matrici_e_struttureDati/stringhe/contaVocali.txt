/**
 * @file conta_vocali.cpp
 * @brief Analizza una stringa inserita dall'utente e conta quante vocali minuscole contiene.
 * @param string testo Il testo inserito dall'utente.
 * @return int Restituisce 0 alla fine dell'esecuzione.
 */

#include  <iostream>
#include <string>

using namespace std;

void conta_vocali(string frase){
    
    int contatore = 0;
    
    for(int i = 0; i < frase.length(); i++){
        if(frase[i] == 'a' || frase[i] == 'e' || frase[i] == 'i' || frase[i] == 'o' || frase[i] == 'u'){
            contatore += 1;
        }
    }
    cout << "Ci sono " << contatore << " vocali" << endl;;

}

int main(){
    
    string frase;
    cout << "Inserisci la frase della quale vuoi contare le vocali: " << endl;
    getline(cin, frase);
    conta_vocali(frase);
    
    return 0;
}
