/**
 * @file analisi_numero.cpp
 * @brief Analisi delle proprietà matematiche di un numero intero.
 * * Il programma richiede all'utente l'inserimento di un numero intero e
 * determina, attraverso strutture condizionali:
 * 1) Se il numero è pari o dispari (usando l'operatore modulo %).
 * 2) Se il numero è positivo, negativo o pari a zero.
 * * @return int Restituisce 0 in caso di esecuzione corretta.
 */
 
 #include <iostream>
 
 using namespace std;
 int main(){
     
     int numero;
     
     cout << "Dato un numero intero verifico se è pari o dispari e positivo, negativo o nessuno dei due" << endl;
     cout << "Che numero vuoi inserire?: ";
     cin >> numero;
     
     //pari o dispari
     if(numero%2 == 0){
         cout << "Il numero è pari" << endl;
     }else{
         cout << "Il numero è dispari" << endl;
     }
     
     //positività
     if(numero == 0){
         cout << "Lo 0 non è ne positio ne negativo" << endl;
     }else if(numero > 0){
         cout << "Il numero è positivo" << endl;
     }else{
         cout << "Il numero è negativo" << endl;
     }
     
     return 0;
 }
