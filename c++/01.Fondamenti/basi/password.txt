/**
 * @file controllo_password.cpp
 * @brief Sistema di autenticazione simulato con tentativi limitati.
 * * Il programma memorizza una password predefinita e concede all'utente
 * un massimo di 3 tentativi per indovinarla. Se l'utente indovina,
 * l'accesso viene consentito; se i tentativi si esauriscono senza successo,
 * il sistema viene bloccato.
 * * @return int Restituisce 0 in caso di esecuzione corretta.
 */
 
 #include <iostream>
 
 using namespace std;
 int main(){
     string passwordSegreta = "lukadoncic";
     string tentativoUtente;
     int contatore = 3;
     int n_tentativi = 0;
     
     cout << "You have 3 attempts, after that the device will explode" << endl;
    
     while(contatore > 0){
         cout << "Insert password: ";
         getline(cin, tentativoUtente);
         
         if( tentativoUtente != passwordSegreta){
             n_tentativi++;
             cout << "Password is incorrect" << endl;
             contatore--;
             
         }
         else{
             n_tentativi++;
             cout << "ACCESS GRANTED, used attempts: " << n_tentativi << endl;
             
             return 0;
         }
         
         
         
    }
    cout << "ACCESS DENIED";
     
     
     return 1;
 }
