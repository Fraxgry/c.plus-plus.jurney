/*********************************************************************************************************
 * GESTIONE DEGLI STATI DINAMICI E GERARCHIE DI ACCESSO
 * * Esercizio sulla modellazione incrementale di dati protetti combinati con liste dinamiche.
 * * 1. CLASSE BASE: "Persona"
 * - Gestione base del nome tramite costruttore standard.
 * * 2. CLASSE FIGLIA: "Atleta" (Eredita da Persona)
 * - Attributi privati (`private`): 'km_corsi' (accumulatore interno gestibile solo tramite regole della classe).
 * - Costruttore a cascata: aggancia il nome al Padre e azzera lo stato dei chilometri.
 * - Metodo `aggiungi_km()`: Incapsulamento puro con validazione dei dati ad incremento (valori strettamente positivi).
 * * NEL MAIN:
 * - Creazione di una squadra tramite `std::vector<Atleta>`.
 * - Algoritmo di scansione combinato con aggiornamento dinamico degli stati interni degli oggetti.
 *********************************************************************************************************/

#include <iostream>
#include <string>
#include <vector>

using namespace std;

class Persona{
    
public:
    string nome;
    
    Persona(string n){
        nome = n;
    }

};

class Atleta : public Persona{
private:
    float km_corsi;
public:
    Atleta(string n, float k) : Persona(n){
        km_corsi = k;
    }
    
    float get_k(){
        return km_corsi;
    }
    
    void aggiungi_km(float km){
        if(km > 0){
            km_corsi += km;
        }
    }
    
};


int main(){
    
    Atleta a1("Alessandro", 0.0);
    Atleta a2("Francesco", 0.0);
    Atleta a3("Gaia", 0.0);
    
    vector<Atleta> squadra;
    squadra.push_back(a1);
    squadra.push_back(a2);
    squadra.push_back(a3);
    
    for(int i = 0; i < squadra.size(); i++){
        cout << squadra[i].nome << " ha persorso: " << squadra[i].get_k() << " km" << endl;
    }
    
    squadra[1].aggiungi_km(8.9);
    cout << "Dopo la modifica." << endl;
    
    for(int i = 0; i < squadra.size(); i++){
        cout << squadra[i].nome << " ha persorso: " << squadra[i].get_k() << " km" << endl;
    }
    
    return 0;
}
