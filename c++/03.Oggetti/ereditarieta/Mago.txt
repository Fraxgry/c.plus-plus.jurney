/*********************************************************************************************************
 * INTERAZIONE DIRETTA E MUTAZIONE DEGLI STATI TRA CLASSI FRATELLE
 * * Esercizio avanzato sulla collaborazione e sul passaggio di oggetti come parametri di metodi.
 * * 1. GERARCHIA DI BASE:
 * - Classe base "Personaggio" che unifica nome e punti vita comuni a tutte le entità del gioco.
 * - Classe derivata "Mago" specializzata con l'attributo 'mana'.
 * - Classe derivata "Guerriero" specializzata con l'attributo 'forza'.
 * * INTERAZIONE TRA OGGETTI (Metodo Cross-Class):
 * - Studio del metodo `attacca(Mago &bersaglio)` all'interno della classe Guerriero.
 * - Passaggio dell'oggetto bersaglio tramite **riferimento diretto** (`&`). Questo permette al Guerriero 
 * di modificare a runtime e in tempo reale i punti vita dell'istanza del Mago presente nel main, 
 * simulando un vero e proprio sistema di combattimento ad oggetti.
 *********************************************************************************************************/

#include <iostream>
#include <string>
#include <vector>

using namespace std;

class Personaggio{
public:
    string nome;
    int vita;
   
    Personaggio(string n, int v){
        nome = n;
        vita = v;
    }
    void mostra_stato(){
        cout << nome << " ha " << vita << " HP." << endl;
       
    }

};

class Mago : public Personaggio{
public:
    int mana;
   
    Mago(string n, int v, int m) :  Personaggio(n,v){
    mana = m;
    }
   
    void lancia_incantesimo(){
        cout << "Mago lancia palla di fuoco" << endl;
    }
};

class Guerriero : public Personaggio{
public:
    int forza;
   
    Guerriero(string n, int v, int f) : Personaggio(n,v){
    forza = f;
    }
   
    void attacca(Mago &bersaglio){
        cout << nome << " attacca con la sua ascia!" << endl;
        bersaglio.vita -= forza;
        if(bersaglio.vita < 0){
            bersaglio.vita = 0;
        }
        cout << bersaglio.vita << endl;
    }
};
   

int main()
{
    vector<Mago> maghi;
    vector<Guerriero> guerrieri;
   
    maghi.push_back(Mago("Gandalf", 80, 150));
    guerrieri.push_back(Guerriero("Thor", 120, 150));
   
    guerrieri[0].attacca(maghi[0]);
   
   
   
    return 0;
}
