 * - Definizione di una classe semplice chiamata "Spada".
 * - Attributi pubblici (`public`): 'nome' (string), 'danno_base' (int), 'durability' (int).
 * - Metodi di azione (`void`): 
 * - 'usa()': riduce l'integrità dell'arma e ne mostra lo stato residuo.
 * - 'affila()': ripristina la durevolezza al valore massimo e applica un bonus permanente al danno.
 * * NEL MAIN:
 * - Creazione e istanziazione di un oggetto statico sullo stack.
 * - Inizializzazione diretta degli attributi tramite l'operatore punto (.).
 * - Chiamata sequenziale dei metodi dell'oggetto.
 *********************************************************************************************************/

#include <iostream>
#include <string>

using namespace std;
class Spada{
public:
    string nome;
    int danno_base;
    int durability;
   
    void usa()
    {
       
        durability -= 5;
        cout << "Used" << nome << "! Durability left: " << durability << endl;
       
    }
   
    void affila()
    {
        if(durability < 100)
        {
            durability = 100;
            danno_base += 2;
            cout << "Durability restored";
        }else{
            cout << "Already max";
        }
       
       
    }
   
};

int main()
{
    Spada spada;
   
    spada.nome = "Excalibur";
    spada.danno_base = 50;
    spada.durability = 100;
   
    spada.usa();
    spada.affila();
   
    return 0;
}
