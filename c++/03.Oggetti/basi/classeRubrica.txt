#include <iostream>
#include <vector>
#include <string>

using namespace std;

class Contatto{
public:
    string nome;
    string numero;
};

class Rubrica{
public:
    vector<Contatto> lista_contatti;
    
    void cerca(string ricercato){
        for(int i = 0; i < lista_contatti.size(); i++){
            if(lista_contatti[i].nome == ricercato){
                cout << lista_contatti[i].numero;
                return;
            }
        }
        cout << "Non presente";
    }
    
    void aggiungi(Contatto nuovo_contatto){
        lista_contatti.push_back(nuovo_contatto);
    }
};

int main(){
    
    Contatto c1;
    Contatto c2;
    Contatto c3;
    c1.nome = "Fra";
    c1.numero = "351 912 1771";
    c2.nome = "GGGDFFD";
    c2.numero = "44355244";
    c3.nome = "gaga";
    c3.numero = "44355244";
    
    Rubrica mia_rubrica;
    mia_rubrica.aggiungi(c1);
    mia_rubrica.aggiungi(c2);
    mia_rubrica.aggiungi(c3);
    mia_rubrica.cerca("gaga");
    
    return 0;
}
